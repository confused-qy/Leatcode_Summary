class Solution {
public:
    vector<int> partitionLabels(string s) {
        // 遍历一遍，记录每个字母最后出现的位置
        unordered_map<char, int> mp;
        for (int i = 0; i < s.size(); i++) {
            mp[s[i]] = i;
        }
        // 再遍历一遍，根据每个字母的末尾位置，寻找每段长度
        vector<int> res;
        int step = -1;
        int begin = 0;
        for (int j = 0; j < s.size(); j++) {
            step = max(step, mp[s[j]] - begin);
            if (j == begin + step) {
                res.push_back(step + 1);
                step = -1;
                begin = j + 1;
            }
        }
        return res;
    }
};

// 提高速度：哈希表可以考虑换成数组
