class Solution {
public:
    vector<string> letterCombinations(string digits) {
        // 定义一一对应的关系
        unordered_map<char, string> phonemap{
            {'2', "abc"}, {'3', "def"},  {'4', "ghi"}, {'5', "jkl"},
            {'6', "mno"}, {'7', "pqrs"}, {'8', "tuv"}, {'9', "wxyz"},
        };

        // 用来存答案的vector
        vector<string> combinations;
        if (digits.length() == 0) {
            return combinations;
        } // 空集情况
        string combination;
        backtrack(combinations, 0, combination, phonemap, digits);
        return combinations;
    }

    // 非常重要：combinations不能按值传递！！要加&！
    void backtrack(vector<string>& combinations, int index, string combination,
                   unordered_map<char, string> phonemap, string digits) {
        if (index == digits.length()) {
            combinations.push_back(combination);
        } else {
            // 获取当前这一位的数字
            char digit = digits[index];
            // 获取这个数字对应哪些字母
            string current = phonemap[digit];
            // 枚举这一位所有可能的字母
            for (int i = 0; i < current.length(); i++) {
                // 取一个字母放入当下的string
                combination.push_back(current[i]);
                // 枚举下一位数字的字母
                backtrack(combinations, index + 1, combination, phonemap,
                          digits);
                // 拿走这个字母，更换另一个情况
                combination.pop_back();
            }
        }
    }
};
